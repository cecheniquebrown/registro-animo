  <script>
    // Configurar fecha m√°xima (hoy) en el date-picker
    document.addEventListener("DOMContentLoaded", function () {
      const fechaInput = document.getElementById("fecha");
      const today = new Date().toISOString().split("T")[0];
      fechaInput.setAttribute("max", today);

      // Cargar endpoint guardado (si existe)
      const savedEndpoint = localStorage.getItem("registro_animo_endpoint");
      const endpointInput = document.getElementById("endpointInput");
      if (savedEndpoint) {
        endpointInput.value = savedEndpoint;
      }
    });

    const endpointInput = document.getElementById("endpointInput");
    const saveEndpointBtn = document.getElementById("saveEndpointBtn");
    const configStatus = document.getElementById("configStatus");
    const selfTestBtn = document.getElementById("selfTestBtn");

    saveEndpointBtn.addEventListener("click", function () {
      const val = (endpointInput.value || "").trim();
      if (!val) {
        configStatus.textContent = "Pega el endpoint antes de guardar.";
        configStatus.className = "status error";
        return;
      }
      localStorage.setItem("registro_animo_endpoint", val);
      configStatus.textContent = "Endpoint guardado. Ya puedes enviar registros desde este navegador.";
      configStatus.className = "status ok";
    });

    selfTestBtn.addEventListener("click", function () {
      const ep = (localStorage.getItem("registro_animo_endpoint") || endpointInput.value || "").trim();
      if (ep && ep.startsWith("http")) {
        configStatus.textContent = "Auto-chequeo: endpoint parece v√°lido.";
        configStatus.className = "status ok";
      } else {
        configStatus.textContent = "Auto-chequeo: falta configurar un endpoint v√°lido.";
        configStatus.className = "status error";
      }
    });

    // Manejo del slider de nota
    const notaInput = document.getElementById("nota");
    const notaLabel = document.getElementById("notaLabel");

    notaInput.addEventListener("input", function () {
      const val = Number(notaInput.value);
      notaLabel.textContent = val === 0 ? "0 = sin evaluar" : String(val);
    });

    // Env√≠o del formulario (ahora como formulario cl√°sico, no JSON)
    const form = document.getElementById("registroForm");
    const formStatus = document.getElementById("formStatus");

    form.addEventListener("submit", async function (e) {
      e.preventDefault();
      formStatus.textContent = "";
      formStatus.className = "status";

      const endpoint = (localStorage.getItem("registro_animo_endpoint") || endpointInput.value || "").trim();
      if (!endpoint) {
        formStatus.textContent = "Falta configurar el endpoint en la parte superior.";
        formStatus.className = "status error";
        return;
      }

      const codigoPaciente = document.getElementById("codigoPaciente").value.trim();
      const fecha = document.getElementById("fecha").value;
      const contexto = document.getElementById("contexto").value.trim();
      const estado = document.getElementById("estado").value.trim();
      const nota = Number(notaInput.value);
      const emociones = document.getElementById("emociones").value.trim();
      const pensamientos = document.getElementById("pensamientos").value.trim();
      const consent = document.getElementById("consent").checked;

      if (!fecha || !estado) {
        formStatus.textContent = "Completa al menos Fecha y Estado de √°nimo.";
        formStatus.className = "status error";
        return;
      }
      if (!consent) {
        formStatus.textContent = "Debes aceptar el consentimiento para enviar.";
        formStatus.className = "status error";
        return;
      }
      if (!Number.isFinite(nota) || nota < 1 || nota > 10) {
        formStatus.textContent = "Selecciona la Nota del d√≠a entre 1 y 10 (mueve el deslizador).";
        formStatus.className = "status error";
        return;
      }

      const payload = {
        codigo_paciente: codigoPaciente,
        fecha,
        contexto,
        estado,
        nota,
        emociones,
        pensamientos,
        consent,
        submitted_at: new Date().toISOString(),
      };

      try {
        formStatus.textContent = "Enviando‚Ä¶";

        // üëâ Enviar como formulario cl√°sico (URL-encoded), sin JSON ni cabecera especial
        const body = new URLSearchParams();
        Object.entries(payload).forEach(([key, value]) => {
          body.append(key, String(value));
        });

        const res = await fetch(endpoint, {
          method: "POST",
          body
        });

        if (!res.ok) {
          const text = await res.text();
          throw new Error(text || "Error al enviar");
        }

        formStatus.textContent = "¬°Registro enviado! Gracias por completar tu d√≠a.";
        formStatus.className = "status ok";

        // Mantener fecha y c√≥digo paciente; limpiar el resto y resetear nota a 0
        document.getElementById("contexto").value = "";
        document.getElementById("estado").value = "";
        document.getElementById("emociones").value = "";
        document.getElementById("pensamientos").value = "";
        document.getElementById("consent").checked = false;
        notaInput.value = 0;
        notaLabel.textContent = "0 = sin evaluar";
      } catch (err) {
        formStatus.textContent = "Error al enviar: " + (err.message || err);
        formStatus.className = "status error";
      }
    });

    // Auto-chequeo del formulario
    const formSelfTestBtn = document.getElementById("formSelfTestBtn");
    formSelfTestBtn.addEventListener("click", function () {
      const messages = [];
      const endpoint = (localStorage.getItem("registro_animo_endpoint") || endpointInput.value || "").trim();
      if (!endpoint || !endpoint.startsWith("http")) {
        messages.push("‚úó Endpoint no configurado o inv√°lido.");
      } else {
        messages.push("‚úì Endpoint configurado.");
      }
      const fecha = document.getElementById("fecha").value;
      const estado = document.getElementById("estado").value.trim();
      if (!fecha || !estado) {
        messages.push("‚úó Falta Fecha o Estado de √°nimo.");
      } else {
        messages.push("‚úì Fecha y Estado de √°nimo presentes.");
      }
      const nota = Number(notaInput.value);
      if (!(Number.isFinite(nota) && nota >= 0 && nota <= 10)) {
        messages.push("‚úó Nota fuera de rango (0‚Äì10).");
      } else {
        messages.push("‚úì Nota en rango (0‚Äì10). 0 = sin evaluar.");
      }
      const consent = document.getElementById("consent").checked;
      if (!consent) {
        messages.push("‚úó Consentimiento no aceptado (es obligatorio para enviar).");
      } else {
        messages.push("‚úì Consentimiento aceptado.");
      }
      formStatus.textContent = messages.join(" | ");
      formStatus.className = messages.some(m => m.startsWith("‚úó")) ? "status error" : "status ok";
    });
